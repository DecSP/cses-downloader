using namespace std;
 
int n,m;
vector <vector<int>> adList;
vector <bool> visit;
vector <bool> marked;
vector<int> path;
	
bool dfs(int curr, vector<int>&eulerPath,int p){
    marked[curr]=true;
    eulerPath.push_back(curr);
    for (int &v:adList[curr]) {
        if (visit[v]) {
            if (p!=v&&marked[v]) {
                path.push_back(v);
                int eusz=eulerPath.size();
                int tmp=eusz-1;
                while (eulerPath[tmp]!=v){
                    path.push_back(eulerPath[tmp]);
                    --tmp;
                }
                path.push_back(v);
                return true;
            }
            continue;
        }
        visit[v]=true;
        if (dfs(v,eulerPath,curr)) return true;
    }
    marked[curr]=false;
    eulerPath.pop_back();
    return false;
}

	fastio;

	adList.assign(n,vector<int>());
    visit.assign(n,0);
    marked.assign(n,0);
    int x,y;
	for (int i=0;i<m;++i){
		cin>>x>>y;x--;y--;
		adList[x].push_back(y);
		adList[y].push_back(x);
	}
    vector<int>eulerPath;
    for (int start=0;start<n;++start){
        if (visit[start]) continue;
        visit[start]=true;
        if (dfs(start,eulerPath,-1)) {
